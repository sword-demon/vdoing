sync包的Pool

# sync.Pool

# 基本用法

# 概念

协程安全、可伸缩的，用于存放可重用对象的容器。

原始目的

存放已分配但是暂时不用的对象，需要时直接从pool里取，然后放回，以减小 GC 回收的压力。

GC 压力是啥

很简单，对象多了，GC 回收自然压力就大

提示

可以去了解一下内存标记

var p *sync.Pool

type User struct {
	Name string
}

func main() {
	p = &sync.Pool{
		New: func() interface{} {
			log.Println("create user")
			return &User{Name: "张三"}
		},
	}
	u1 := p.Get().(*User)
	fmt.Println(u1)
	u1.Name = "李四"
	p.Put(u1)
	u2 := p.Get()
	fmt.Println(u2)
}

// ServeHTTP conforms to the http.Handler interface.
func (engine *Engine) ServeHTTP(w http.ResponseWriter, req *http.Request) {
	c := engine.pool.Get().(*Context)
	c.writermem.reset(w)
	c.Request = req
	c.reset()

	engine.handleHTTPRequest(c)

	engine.pool.Put(c)
}
